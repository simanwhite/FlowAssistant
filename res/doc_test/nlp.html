if { [string equal $env(ENV_VIEW) default] } {
    setenv DJ_CONTEXT kryptos

    setenv DJ_CONTEXT kryptos_gio_only

} elseif { [string equal $env(ENV_VIEW) kryptos_lite] } {
    setenv DJ_CONTEXT kryptos_lite
++++++++++++++++++++++++++++++++++++++++++++++++
} elseif { [string equal $env(ENV_VIEW) nlp] } {
    setenv DJ_CONTEXT kryptos
++++++++++++++++++++++++++++++++++++++++++++++++
} else {
    reportWarning "ERROR: Unknown view setting: $env(ENV_VIEW)"
    reportWarning "ERROR: Bootenv failed"
    exit 1
}
