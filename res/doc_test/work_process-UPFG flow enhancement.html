Solution proposal 1

Use xml to descript PST:
<PST>
	<block_names>block1, block2, block3</block_names>
	
	<power_state>
		<state_name>P1_OFF</state_name>
		<state_list>OFF, ON, ON</state_list>
	</power_state>
	
	<power_state>
		<state_name>P2_OFF</state_name>
		<state_list>ON, OFF, ON</state_list>
	</power_state>
	
	<power_state>
		<state_name>P3_OFF</state_name>
		<state_list>ON, ON, OFF</state_list>
	</power_state>
	.
	.
	.
</PST>
